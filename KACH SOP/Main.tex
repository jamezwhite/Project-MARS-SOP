\makeatletter

\makeatother

% Put ONLY the chapter title (not "Chapter 1") in the running header mark
\renewcommand{\chaptermark}[1]{\markboth{#1}{}}

% --- Header: top-left shows chapter title on non-opening pages
\fancyhead[L]{\nouppercase{\leftmark}}

% --- Footer: left = fixed text + front-page date, right = page number
\fancyfoot[L]{Project MARS SOP, \thedate}

% Optional rules (lines). Header rule on; footer rule off.
\renewcommand{\headrulewidth}{0.4pt}

% Make chapter opening pages use a blank header but keep the same footer
% (report class sets chapter openings to \thispagestyle{plain} by default)
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[L]{Project MARS SOP, \thedate}%
  \fancyfoot[R]{\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Paragraph Numbering %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Chapter and Section numbering formats
\renewcommand{\thechapter}{\arabic{chapter}}
\renewcommand{\thesection}{\thechapter-\arabic{section}}

% Paragraph / Subparagraph counters and print formats
\makeatletter
\@addtoreset{paragraph}{section}% reset a., b., c. each new section
\@addtoreset{subparagraph}{paragraph}% reset (1), (2) under each letter
\makeatother

\renewcommand{\theparagraph}{\alph{paragraph}}
\renewcommand{\thesubparagraph}{(\arabic{subparagraph})}

% Label styles with titlesec (run-in headings)
\titleformat{\section}
  {\normalfont\Large\bfseries}{\thesection}{1em}{}

% Paragraph: label = italic a., title = bold
\titleformat{\paragraph}[runin]
  {\normalfont\normalsize}%
  {\textit{\theparagraph.}}{0.75em}{\bfseries}[\normalfont]

% Subparagraph: label = (1), title = italics
\titleformat{\subparagraph}[runin]
  {\normalfont\normalsize}%
  {\thesubparagraph}{0.75em}{\itshape}[\normalfont]

% Optional: nicer spacing for the run-in heads
\titlespacing*{\paragraph}{0pt}{0.5\baselineskip}{0.75em}
\titlespacing*{\subparagraph}{0pt}{0pt}{0.75em}
